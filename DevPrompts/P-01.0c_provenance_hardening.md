### Шаг 1: Implement Immutable Provenance Dataclass

Centralize provenance computation в **single point of truth**. Existing helpers в `src/py/rca/provenance.py` upgrade.

**File:** `src/py/rca/provenance.py`

```python
"""
Centralized provenance computation для Run lifecycle.

DNA §2.1.12 compliance: каждый Run produces immutable Provenance object 
that flows через STARTED log → submission → SUCCEEDED log → asset metadata.

Key invariant: same config dict ALWAYS produces same Provenance object.
Hash drift impossible if Provenance computed once и passed by reference.
"""

from dataclasses import dataclass, field, asdict
from typing import Dict, Any
import hashlib
import json
from datetime import date


@dataclass(frozen=True)  # IMMUTABLE
class Provenance:
    """
    Immutable provenance bundle для one Run.
    
    Computed once via compute_provenance(config) at process start.
    Pass by reference к all logging + asset metadata operations.
    
    Frozen=True prevents accidental mutation downstream.
    """
    config_id: str          # human-readable preset name (e.g., "default")
    params_hash: str        # SHA-256 hex of canonical_serialize(config)
    run_id: str             # f"{config_id}_{period}_{params_hash[:8]}"
    config_serialized: str  # canonical JSON for full audit trail
    algorithm_version: str
    rna_version: str
    build_date: str         # ISO date
    
    def to_asset_properties(self) -> Dict[str, Any]:
        """
        Format Provenance fields для setAssetProperties API.
        
        Excludes config_serialized (too large for asset properties).
        Full config preserved в logs/runs.jsonl.
        """
        return {
            'config_id': self.config_id,
            'params_hash': self.params_hash,
            'run_id': self.run_id,
            'algorithm_version': self.algorithm_version,
            'rna_version': self.rna_version,
            'build_date': self.build_date
        }
    
    def to_log_entry(self, lifecycle_event: str, **extra) -> Dict[str, Any]:
        """
        Format Provenance for logs/runs.jsonl entry.
        
        lifecycle_event: "STARTED" | "SUCCEEDED" | "FAILED"
        extra: any additional event-specific fields
        """
        return {
            'event': lifecycle_event,
            'run_id': self.run_id,
            'params_hash': self.params_hash,
            'config_id': self.config_id,
            'config_serialized': self.config_serialized,
            'algorithm_version': self.algorithm_version,
            'rna_version': self.rna_version,
            'build_date': self.build_date,
            'event_timestamp': date.today().isoformat(),
            **extra
        }


def canonical_serialize(config: Dict[str, Any]) -> str:
    """
    Deterministic JSON serialization для config dict.
    
    sort_keys=True ensures dict ordering doesn't affect hash.
    separators=(',', ':') eliminates whitespace variations.
    
    THIS IS THE ONLY ALLOWED SERIALIZATION для params_hash computation.
    Any other serialization will produce different hashes.
    """
    return json.dumps(config, sort_keys=True, separators=(',', ':'))


def compute_provenance(
    config: Dict[str, Any],
    config_id: str,
    period: str,
    algorithm_version: str = "2.3",
    rna_version: str = "1.2"
) -> Provenance:
    """
    Compute immutable Provenance object.
    
    Call ONCE at process start. Pass returned object к all subsequent 
    operations (logging, submission, asset metadata).
    
    Args:
        config: full config dict (e.g., from DEFAULT_PRESET)
        config_id: human-readable preset name
        period: temporal scope identifier (e.g., "2019_2025")
        algorithm_version, rna_version: version strings
    
    Returns:
        Frozen Provenance dataclass
    
    Raises:
        ValueError if config not JSON-serializable
    """
    serialized = canonical_serialize(config)
    params_hash = hashlib.sha256(serialized.encode('utf-8')).hexdigest()
    run_id = f"{config_id}_{period}_{params_hash[:8]}"
    
    return Provenance(
        config_id=config_id,
        params_hash=params_hash,
        run_id=run_id,
        config_serialized=serialized,
        algorithm_version=algorithm_version,
        rna_version=rna_version,
        build_date=date.today().isoformat()
    )
```

```python
import pytest
from src.py.rca.provenance import compute_provenance, canonical_serialize, Provenance


def test_same_config_produces_same_hash():
    """Critical invariant: same config → same Provenance."""
    config = {'key1': 'value', 'key2': 42, 'nested': {'a': 1}}
    
    p1 = compute_provenance(config, 'default', '2019_2025')
    p2 = compute_provenance(config, 'default', '2019_2025')
    
    assert p1.params_hash == p2.params_hash
    assert p1.run_id == p2.run_id


def test_different_dict_order_produces_same_hash():
    """sort_keys=True ensures order-independence."""
    config_a = {'a': 1, 'b': 2}
    config_b = {'b': 2, 'a': 1}
    
    p_a = compute_provenance(config_a, 'default', '2019_2025')
    p_b = compute_provenance(config_b, 'default', '2019_2025')
    
    assert p_a.params_hash == p_b.params_hash


def test_provenance_immutable():
    """Frozen dataclass prevents mutation."""
    config = {'k': 'v'}
    p = compute_provenance(config, 'default', '2019_2025')
    
    with pytest.raises(Exception):  # FrozenInstanceError или AttributeError
        p.params_hash = 'mutated'


def test_to_asset_properties_excludes_config_serialized():
    """Asset properties small; config_serialized stays in logs."""
    p = compute_provenance({'k': 'v'}, 'default', '2019_2025')
    props = p.to_asset_properties()
    
    assert 'config_serialized' not in props
    assert 'params_hash' in props
    assert 'run_id' in props


def test_to_log_entry_includes_all():
    """Log entries get full audit trail."""
    p = compute_provenance({'k': 'v'}, 'default', '2019_2025')
    entry = p.to_log_entry('STARTED', extra_field='value')
    
    assert entry['event'] == 'STARTED'
    assert 'config_serialized' in entry
    assert entry['extra_field'] == 'value'


def test_run_id_format():
    """run_id = config_id_period_hash[:8]"""
    p = compute_provenance({'k': 'v'}, 'default', '2019_2025')
    
    assert p.run_id.startswith('default_2019_2025_')
    assert len(p.run_id.split('_')[-1]) == 8


def test_canonical_serialize_no_whitespace():
    """No whitespace variations possible."""
    s = canonical_serialize({'a': 1, 'b': 2})
    
    assert ' ' not in s
    assert '\n' not in s
```

### Шаг 3: Audit Tool с Allowlist Mechanism

**File:** `tools/audit_provenance_consistency.py`

```python
"""
Audit tool checking provenance consistency across assets и logs.

Exits 0 if all matches OR mismatches in allowlist.
Exits 1 if unexpected mismatches detected.

Run: python tools/audit_provenance_consistency.py
CI integration: blocks merge if unexpected mismatches.
"""

import ee
import json
from pathlib import Path
from typing import Dict, List, Tuple
import sys


ASSET_ROOT = 'projects/nodal-thunder-481307-u1/assets/RuPlumeScan'
LOGS_PATH = Path('logs/runs.jsonl')
ALLOWLIST_PATH = Path('tools/provenance_audit_allowlist.json')


def load_allowlist() -> List[Dict]:
    """Load known mismatches that are tracked для remediation."""
    if not ALLOWLIST_PATH.exists():
        return []
    return json.loads(ALLOWLIST_PATH.read_text(encoding='utf-8'))['known_mismatches']


def load_runs_log() -> List[Dict]:
    """Load all entries from logs/runs.jsonl."""
    if not LOGS_PATH.exists():
        return []
    
    entries = []
    for line in LOGS_PATH.read_text(encoding='utf-8').splitlines():
        if line.strip():
            entries.append(json.loads(line))
    return entries


def get_asset_provenance(asset_id: str) -> Dict:
    """Read provenance properties from asset metadata."""
    try:
        info = ee.data.getAsset(asset_id)
        props = info.get('properties', {})
        return {
            'asset_id': asset_id,
            'config_id': props.get('config_id'),
            'params_hash': props.get('params_hash'),
            'run_id': props.get('run_id'),
            'has_provenance': all(k in props for k in ['config_id', 'params_hash', 'run_id'])
        }
    except Exception as e:
        return {'asset_id': asset_id, 'error': str(e), 'has_provenance': False}


def list_baseline_assets() -> List[str]:
    """Enumerate все baseline + catalog assets."""
    parents = [
        f'{ASSET_ROOT}/baselines',
        f'{ASSET_ROOT}/catalog/CH4',
        f'{ASSET_ROOT}/catalog/NO2',
        f'{ASSET_ROOT}/catalog/SO2'
    ]
    
    asset_ids = []
    for parent in parents:
        try:
            response = ee.data.listAssets({'parent': parent})
            for asset in response.get('assets', []):
                # Skip _temp/ and other helper folders
                if '_temp' not in asset['name']:
                    asset_ids.append(asset['name'])
        except Exception:
            pass  # parent may not exist yet
    
    return asset_ids


def is_allowlisted(asset_id: str, issue: str, allowlist: List[Dict]) -> bool:
    """Check if mismatch is known and tracked."""
    asset_basename = asset_id.split('/')[-1]
    for entry in allowlist:
        if entry['asset'].endswith(asset_basename):
            return True
    return False


def audit() -> Tuple[List[Dict], List[Dict]]:
    """
    Run audit. Returns (unexpected_mismatches, allowlisted_mismatches).
    
    Exits 1 if unexpected_mismatches не пуст.
    """
    ee.Initialize(project='nodal-thunder-481307-u1')
    
    allowlist = load_allowlist()
    runs_log = load_runs_log()
    asset_ids = list_baseline_assets()
    
    unexpected = []
    allowlisted = []
    
    # Build run_id → log entries mapping
    runs_by_id = {}
    for entry in runs_log:
        run_id = entry.get('run_id')
        if run_id:
            runs_by_id.setdefault(run_id, []).append(entry)
    
    for asset_id in asset_ids:
        asset_prov = get_asset_provenance(asset_id)
        asset_basename = asset_id.split('/')[-1]
        
        # Check 1: asset has provenance fields
        if not asset_prov.get('has_provenance'):
            issue = f'Asset missing provenance fields'
            if is_allowlisted(asset_id, issue, allowlist):
                allowlisted.append({'asset': asset_id, 'issue': issue})
            else:
                unexpected.append({'asset': asset_id, 'issue': issue})
            continue
        
        # Check 2: asset hash matches log entries
        run_id = asset_prov['run_id']
        if run_id in runs_by_id:
            log_hashes = set(e.get('params_hash') for e in runs_by_id[run_id])
            asset_hash = asset_prov['params_hash']
            
            if asset_hash not in log_hashes:
                issue = f'Asset hash {asset_hash[:8]} ≠ log hashes {[h[:8] for h in log_hashes]}'
                if is_allowlisted(asset_id, issue, allowlist):
                    allowlisted.append({'asset': asset_id, 'issue': issue})
                else:
                    unexpected.append({'asset': asset_id, 'issue': issue})
        # If no log entries — that's separately a CR concern but not blocker here
    
    return unexpected, allowlisted


def main():
    print('=' * 60)
    print('PROVENANCE CONSISTENCY AUDIT')
    print('=' * 60)
    
    unexpected, allowlisted = audit()
    
    print(f'\nAllowlisted mismatches ({len(allowlisted)}):')
    for entry in allowlisted:
        print(f'  - {entry["asset"]}: {entry["issue"]}')
    
    print(f'\nUnexpected mismatches ({len(unexpected)}):')
    if not unexpected:
        print('  (none)')
    for entry in unexpected:
        print(f'  - {entry["asset"]}: {entry["issue"]}')
    
    print('\n' + '=' * 60)
    if unexpected:
        print('AUDIT FAILED — unexpected provenance mismatches detected')
        sys.exit(1)
    else:
        print('AUDIT PASSED')
        sys.exit(0)


if __name__ == '__main__':
    main()
```
